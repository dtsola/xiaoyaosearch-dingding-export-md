### 5.2 核心Composable设计

#### useDingTalkAPI.ts

```typescript
export function useDingTalkAPI() {
  const loading = ref(false);
  const error = ref<Error | null>(null);

  // 获取文档列表
  const getDocList = async (dentryUuid: string) => {
    loading.value = true;
    try {
      const result = await dingtalkAPI.getDocList(dentryUuid);
      return result;
    } catch (e) {
      error.value = e as Error;
      throw e;
    } finally {
      loading.value = false;
    }
  };

  // 导出文档
  const exportDoc = async (
    dentryUuid: string,
    format: ExportFormat,
    onProgress?: (progress: number) => void
  ) => {
    // 实现导出逻辑
  };

  return {
    loading,
    error,
    getDocList,
    exportDoc,
  };
}
```

#### useExportManager.ts

```typescript
export function useExportManager() {
  const queue = ref<ExportTask[]>([]);
  const isExporting = computed(() => queue.value.some(t => t.status === 'exporting'));

  // 添加导出任务
  const addTask = (file: DentryItem) => {
    const task: ExportTask = {
      id: generateId(),
      file,
      status: 'pending',
      progress: 0,
    };
    queue.value.push(task);
    processQueue();
  };

  // 处理导出队列
  const processQueue = async () => {
    const pendingTasks = queue.value.filter(t => t.status === 'pending');
    const concurrent = getConcurrentLimit();

    for (const task of pendingTasks.slice(0, concurrent)) {
      exportTask(task);
    }
  };

  return {
    queue,
    isExporting,
    addTask,
    retryTask,
    removeTask,
  };
}
```

#### useConfig.ts

```typescript
export function useConfig() {
  const config = useStorage<AppConfig>('dingtalk-export:config', {
    exportFormats: {
      adoc: 'md',
      axls: 'xlsx',
      adraw: 'jpg',
      amind: 'jpg',
    },
    theme: 'auto',
    autoClose: false,
    concurrentLimit: 3,
  });

  const getFormat = (fileType: string) => {
    return config.value.exportFormats[fileType] || 'md';
  };

  const setFormat = (fileType: string, format: string) => {
    config.value.exportFormats[fileType] = format;
  };

  return {
    config,
    getFormat,
    setFormat,
  };
}
```

## 7. 技术风险

| 风险 | 影响 | 应对方案 |
|------|------|---------|
| **钉钉API变更** | 高 | • 参考老项目稳定的API封装<br>• 模块化设计，降低API变更成本<br>• 建立用户反馈机制，快速响应<br>• 定期测试验证功能 |

| **内存泄漏** | 中 | • 使用 Vue 3 响应式系统，自动清理<br>• 正确使用组合式函数生命周期<br>• 定期进行性能测试 |
