### 3.1 Baby 框架实现

创建 `src/script/index.js`：

```javascript
// Baby 框架核心 - 轻量级响应式框架
function reactive(data) {
    // 基于 Object.defineProperty 的数据劫持
    const keys = Object.keys(data);
    keys.forEach(key => {
        let value = data[key];
        Object.defineProperty(data, key, {
            get() {
                return value;
            },
            set(newValue) {
                if (value !== newValue) {
                    value = newValue;
                    // 触发视图更新
                    if (this._component) {
                        this._component.update();
                    }
                }
            }
        });
    });
    return data;
}

// 虚拟 DOM 创建函数
function h(tag, props, children) {
    return { tag, props, children };
}

// 组件基类
class Component {
    constructor(options) {
        this.data = options.data || {};
        this.methods = options.methods || {};
        this._render = options.render;
        
        // 生命周期钩子
        if (options.created) {
            options.created.call(this);
        }
        
        // 响应式数据
        reactive(this.data);
        this.data._component = this;
        
        // 绑定方法
        Object.keys(this.methods).forEach(key => {
            this[key] = this.methods[key].bind(this);
        });
    }
    
    mount() {
        if (this._render) {
            const vnode = this._render.call(this, h);
            this.el = this.createElement(vnode);
            if (this._mountTarget) {
                this._mountTarget.appendChild(this.el);
            }
        }
        if (this.mounted) {
            this.mounted.call(this);
        }
    }
    
    createElement(vnode) {
        if (typeof vnode === 'string' || typeof vnode === 'number') {
            return document.createTextNode(vnode);
        }
        
        const el = document.createElement(vnode.tag);
        
        // 设置属性
        if (vnode.props) {
            Object.keys(vnode.props).forEach(key => {
                if (key === 'class') {
                    el.className = vnode.props[key];
                } else if (key === 'style') {
                    Object.assign(el.style, vnode.props[key]);
                } else if (key.startsWith('on')) {
                    const event = key.substring(2).toLowerCase();
                    el.addEventListener(event, vnode.props[key]);
                } else {
                    el.setAttribute(key, vnode.props[key]);
                }
            });
        }
        
        // 处理子节点
        if (Array.isArray(vnode.children)) {
            vnode.children.forEach(child => {
                el.appendChild(this.createElement(child));
            });
        } else if (vnode.children) {
            el.appendChild(this.createElement(vnode.children));
        }
        
        return el;
    }
    
    update() {
        // 简化版更新：重新渲染
        if (this._render && this.el) {
            const newEl = this.createElement(this._render.call(this, h));
            this.el.parentNode.replaceChild(newEl, this.el);
            this.el = newEl;
        }
    }
}

// 创建应用实例
function Baby(options) {
    const component = new Component(options);
    return component;
}

export default Baby;
```

### 3.2 HTTP 请求封装

创建 `src/script/Http.js`：

```javascript
/**
 * HTTP 请求封装（基于 XHR）
 * 支持进度监控和自定义请求头
 */
export default function httpRequest(options) {
    return new Promise((resolve, reject) => {
        const xhr = new XMLHttpRequest();
        
        // 设置超时
        xhr.timeout = options.timeout || 30000;
        
        // 设置请求头
        if (options.headers) {
            Object.keys(options.headers).forEach(key => {
                xhr.setRequestHeader(key, options.headers[key]);
            });
        }
        
        // 监听上传进度
        if (options.uploadProgress) {
            xhr.upload.addEventListener('progress', (e) => {
                if (e.lengthComputable) {
                    options.uploadProgress(e.loaded, e.total);
                }
            });
        }
        
        // 监听下载进度
        if (options.downloadProgress) {
            xhr.addEventListener('progress', (e) => {
                if (e.lengthComputable) {
                    options.downloadProgress(e.loaded, e.total);
                }
            });
        }
        
        // 监听响应
        xhr.addEventListener('load', () => {
            if (xhr.status >= 200 && xhr.status < 300) {
                const response = options.responseType === 'json' 
                    ? JSON.parse(xhr.responseText) 
                    : xhr.response;
                if (options.onEnd) {
                    options.onEnd(null, response);
                }
                resolve(response);
            } else {
                const error = new Error(xhr.statusText);
                error.status = xhr.status;
                if (options.onEnd) {
                    options.onEnd(error, null);
                }
                reject(error);
            }
        });
        
        // 监听错误
        xhr.addEventListener('error', () => {
            const error = new Error('Network error');
            if (options.onEnd) {
                options.onEnd(error, null);
            }
            reject(error);
        });
        
        // 监听超时
        xhr.addEventListener('timeout', () => {
            const error = new Error('Request timeout');
            if (options.onEnd) {
                options.onEnd(error, null);
            }
            reject(error);
        });
        
        // 发送请求
        xhr.open(options.method || 'GET', options.url);
        
        if (options.data) {
            xhr.send(options.data);
        } else {
            xhr.send();
        }
    });
}
```

```javascript
/**
 * 配置管理（基于 localStorage）
 */
export const CFG_KEY = {
    EXPORT_ADOC_AS: "export_adoc_as",      // 文档导出格式
    EXPORT_AXLS_AS: "export_axls_as",      // 表格导出格式
    EXPORT_ADRAW_AS: "export_adraw_as",    // 白板导出格式
    EXPORT_AMIND_AS: "export_amind_as"     // 脑图导出格式
};

// 默认配置
const DEFAULT_CONFIG = {
    [CFG_KEY.EXPORT_ADOC_AS]: "docx",
    [CFG_KEY.EXPORT_AXLS_AS]: "xlsx",
    [CFG_KEY.EXPORT_ADRAW_AS]: "jpg",
    [CFG_KEY.EXPORT_AMIND_AS]: "jpg"
};

/**
 * 获取配置
 * @param {string} key 配置键
 * @returns {string} 配置值
 */
export function getCfg(key) {
    const storageKey = `dddd-${key}`;
    const value = localStorage.getItem(storageKey);
    return value !== null ? value : (DEFAULT_CONFIG[key] || "");
}

/**
 * 设置配置
 * @param {string} key 配置键
 * @param {string} value 配置值
 */
export function setCfg(key, value) {
    const storageKey = `dddd-${key}`;
    localStorage.setItem(storageKey, value);
}

/**
 * 获取所有配置
 * @returns {object} 配置对象
 */
export function getAllCfg() {
    const config = {};
    Object.keys(CFG_KEY).forEach(key => {
        config[key] = getCfg(CFG_KEY[key]);
    });
    return config;
}
```

```javascript
 * Chrome 扩展内容脚本
 * 注入到钉钉文档页面，负责初始化应用

// 监听来自 background script 的消息
chrome.runtime.onMessage.addListener((request, sender, sendResponse) => {
    if (request.action === 'showDownloader') {
        // 动态加载 app.js
        initApp();
        sendResponse({ status: 'ok' });
    } else if (request.action === 'showError') {
        alert(request.message);
        sendResponse({ status: 'ok' });
    }
    return true; // 保持消息通道开放
});

/**
 * 初始化应用
 */
async function initApp() {
    // 检查是否已加载
    if (document.querySelector('.my-dingdocdownloader')) {
        return;
    }
    
    try {
        // 动态导入 app.js
        const script = document.createElement('script');
        script.src = chrome.runtime.getURL('assets/app.js');
        script.onload = () => {
            script.remove();
        };
        (document.head || document.documentElement).appendChild(script);
    } catch (error) {
        console.error('Failed to initialize app:', error);
    }
}
```

```javascript
                    if (index > -1) {
                        this.data.selecteds.splice(index, 1);
                    } else {
                        this.data.selecteds.push(doc.id);
                        // 如果是文件夹，递归选择子项
                        if (doc.type === 'folder') {
                            this.selectChildren(doc);
                        }
                    }
                },
                
                selectChildren(doc) {
                    // 递归选择子文件夹中的所有项
                    if (doc.children) {
                        doc.children.forEach(child => {
                            if (!this.data.selecteds.includes(child.id)) {
                                this.data.selecteds.push(child.id);
                            }
                            if (child.type === 'folder') {
                                this.selectChildren(child);
                            }
                        });
                    }
                },
                
                async onDownloadClick() {
                    if (this.data.selecteds.length === 0) {
                        alert('请先选择要导出的文档');
                        return;
                    }
                    
                    try {
                        // 选择保存目录
                        const dirHandle = await window.showDirectoryPicker();
                        
                        // 开始下载
                        for (const docId of this.data.selecteds) {
                            const doc = this.findDoc(this.data.docs, docId);
                            if (doc) {
                                await this.downloadDoc(doc, dirHandle);
                            }
                        }
                        
                        alert(`成功导出 ${this.data.selecteds.length} 个文档`);
                    } catch (error) {
                        if (error.name !== 'AbortError') {
                            console.error('Download failed:', error);
                            alert('下载失败：' + error.message);
                        }
                    }
                },
                
                findDoc(docs, id) {
                    for (const doc of docs) {
                        if (doc.id === id) return doc;
                        if (doc.children) {
                            const found = this.findDoc(doc.children, id);
                            if (found) return found;
                        }
                    }
                    return null;
                },
                
                async downloadDoc(doc, dirHandle) {
                    // 获取导出格式
                    const format = getExportFormat(doc.type);
                    
                    // 调用下载 API
                    await downloadDocument(doc.id, format, dirHandle);
                },
                
                onRefreshClick() {
                    this.loadDocs();
                }
            },
            render(h) {
                return h("div", {class: "main-container"}, [
                    // 进度条
                    this.data.loading ? h("div", {class: "dddd-progress"}, [
                        h("div", {class: "dddd-progress-bar", style: {width: this.data.progress + "%"}})
                    ]) : null,
                    
                    // 文档列表
                    h("ul", {class: "doc-list"}, 
                        this.data.docs.map(doc => 
                            this.renderDocItem(h, doc)
                        )
                    ),
                    
                    // 操作按钮
                    h("div", {class: "action-buttons"}, [
                        h("button", {
                            class: "dddd-btn dddd-btn-ghost",
                            onClick: this.onRefreshClick.bind(this)
                        }, "重新加载"),
                        h("button", {
                            class: "dddd-btn dddd-btn-primary",
                            onClick: this.onDownloadClick.bind(this),
                            disabled: this.data.selecteds.length === 0
                        }, `下载选中 (${this.data.selecteds.length})`)
                    ])
                ]);
            }
        });
        
        this.target = options.target;
    }
    
    mount() {
        super.mount();
        // 加载文档列表
        this.loadDocs();
    }
    
    renderDocItem(h, doc) {
        const isSelected = this.data.selecteds.includes(doc.id);
        
        return h("li", {
            class: "doc-item",
            onClick: () => this.toggleSelect(doc)
        }, [
            h("input", {
                type: "checkbox",
                class: "doc-item-checkbox",
                checked: isSelected,
                readOnly: true
            }),
            h("span", {class: "doc-item-icon"}, 
                doc.type === 'folder' ? '📁' : '📄'
            ),
            h("span", {class: "doc-item-name"}, doc.name),
            doc.size ? h("span", {class: "doc-item-size"}, formatSize(doc.size)) : null
        ]);
    }
}

function getExportFormat(docType) {
    switch (docType) {
        case 'adoc': return getCfg(CFG_KEY.EXPORT_ADOC_AS);
        case 'axls': return getCfg(CFG_KEY.EXPORT_AXLS_AS);
        case 'adraw': return getCfg(CFG_KEY.EXPORT_ADRAW_AS);
        case 'amind': return getCfg(CFG_KEY.EXPORT_AMIND_AS);
        default: return null;
    }
}

function formatSize(bytes) {
    if (bytes < 1024) return bytes + ' B';
    if (bytes < 1024 * 1024) return (bytes / 1024).toFixed(1) + ' KB';
    return (bytes / (1024 * 1024)).toFixed(1) + ' MB';
}
```
