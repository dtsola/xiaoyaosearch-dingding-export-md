### 2.2 Content Script页面检测

**步骤1：实现页面类型检测**

在 `src/content/` 目录下创建 `pageDetector.ts`：

```typescript
export enum PageType {
  MY_SPACE = 'my_space',       // 我的空间
  SPACE = 'space',             // 指定空间
  DOCUMENT = 'document',       // 文档/目录
  UNKNOWN = 'unknown'          // 未知页面
}

export function detectPageType(): PageType {
  const url = window.location.href;
  const pathname = window.location.pathname;

  // 检测"我的空间"页面
  if (pathname.includes('/workspace/mySpace')) {
    return PageType.MY_SPACE;
  }

  // 检测"指定空间"页面
  if (pathname.includes('/workspace/space')) {
    return PageType.SPACE;
  }

  // 检测"文档/目录"页面
  if (pathname.includes('/doc/') || pathname.includes('/docnew/')) {
    return PageType.DOCUMENT;
  }

  return PageType.UNKNOWN;
}

export function getCurrentSpaceId(): string | null {
  const url = new URL(window.location.href);
  const spaceId = url.searchParams.get('spaceId');
  return spaceId;
}

export function getCurrentDocId(): string | null {
  const url = new URL(window.location.href);
  const paths = url.pathname.split('/');
  const docIndex = paths.findIndex(p => p === 'doc' || p === 'docnew');
  
  if (docIndex !== -1 && paths[docIndex + 1]) {
    return paths[docIndex + 1];
  }
  
  return null;
}
```

```typescript
import { defineStore } from 'pinia';
import { ref } from 'vue';

export const useConfigStore = defineStore('config', () => {
  const exportFormats = ref({
    adoc: 'md',      // 文档导出格式
    axls: 'xlsx',    // 表格导出格式
    adraw: 'jpg',    // 白板导出格式
    amind: 'jpg',    // 脑图导出格式
  });

  const theme = ref<'light' | 'dark' | 'auto'>('auto');
  const autoClose = ref(false);
  const concurrentLimit = ref(3);

  function getFormat(fileType: string): string {
    return exportFormats.value[fileType] || 'md';
  }

  function setFormat(fileType: string, format: string): void {
    exportFormats.value[fileType] = format;
    saveConfig();
  }

  function saveConfig(): void {
    localStorage.setItem('dingtalk-export:config', JSON.stringify({
      exportFormats: exportFormats.value,
      theme: theme.value,
      autoClose: autoClose.value,
      concurrentLimit: concurrentLimit.value,
    }));
  }

  function loadConfig(): void {
    const saved = localStorage.getItem('dingtalk-export:config');
    if (saved) {
      try {
        const config = JSON.parse(saved);
        exportFormats.value = config.exportFormats || exportFormats.value;
        theme.value = config.theme || theme.value;
        autoClose.value = config.autoClose ?? autoClose.value;
        concurrentLimit.value = config.concurrentLimit || concurrentLimit.value;
      } catch (e) {
        console.error('加载配置失败:', e);
      }
    }
  }

  // 初始化时加载配置
  loadConfig();

  return {
    exportFormats,
    theme,
    autoClose,
    concurrentLimit,
    getFormat,
    setFormat,
    saveConfig,
  };
});
```

**步骤2：封装钉钉API（完整实现）**

在 `src/sidepanel/api/dingtalk.ts` 中创建API封装：

```typescript
import type { DentryItem, ExportJobResponse, ExportFormat } from './types';

export class DingTalkAPI {
  private baseURL = 'https://api.dingtalk.com';

  /**
   * 获取文档列表
   * 参考老项目 api.js 中的 getDocList 方法
   */
  async getDocList(dentryUuid: string, pageToken?: string): Promise<{
    items: DentryItem[];
    has_more: boolean;
    page_token: string;
  }> {
    // 从当前页面获取lwpClient
    const lwpClient = (window as any).lwpClient;
    if (!lwpClient) {
      throw new Error('未找到钉钉客户端，请确保在钉钉页面使用');
    }

    // 调用钉钉API获取文档列表
    const response = await lwpClient.request({
      url: '/api/doc/list',
      method: 'POST',
      data: {
        dentry_uuid: dentryUuid,
        page_token: pageToken || '',
      },
    });

    return response;
  }

  /**
   * 导出文档为 .docx / .xlsx / .pdf
   * 参考老项目 api.js 中的 downloadAdoc、downloadAxls 方法
   */
  async exportDocument(
    dentryUuid: string,
    format: 'docx' | 'xlsx' | 'pdf',
    onProgress?: (progress: number) => void
  ): Promise<string> {
    const lwpClient = (window as any).lwpClient;

    // 步骤1: upload_info - 上传文档元数据
    const uploadInfo = await lwpClient.request({
      url: '/api/doc/export_upload_info',
      method: 'POST',
      data: {
        dentry_uuid: dentryUuid,
        export_format: format,
      },
    });

    // 步骤2: submitExportJob - 创建导出任务
    const job = await lwpClient.request({
      url: '/api/doc/submit_export_job',
      method: 'POST',
      data: {
        export_info: uploadInfo,
      },
    });

    // 步骤3: queryExportJobInfo - 轮询任务状态
    let status = 'pending';
    let downloadUrl = '';
    const pollInterval = 2000; // 每2秒轮询一次

    while (status === 'pending' || status === 'processing') {
      const result = await lwpClient.request({
        url: '/api/doc/query_export_job_info',
        method: 'POST',
        data: {
          job_id: job.job_id,
        },
      });

      status = result.status;
      
      if (status === 'success') {
        downloadUrl = result.download_url;
        if (onProgress) onProgress(100);
        break;
      } else if (status === 'failed') {
        throw new Error(result.error_message || '导出失败');
      }

      // 等待后继续轮询
      await new Promise(resolve => setTimeout(resolve, pollInterval));
      
      if (onProgress) {
        const progress = this.calculateProgress(result);
        onProgress(progress);
      }
    }

    return downloadUrl;
  }

  /**
   * 导出文档为 PDF
   * 参考老项目 api.js 中的 downloadToPDF 方法
   */
  async exportToPDF(dentryUuid: string): Promise<string> {
    const lwpClient = (window as any).lwpClient;

    // 步骤1: upload_info
    const uploadInfo = await lwpClient.request({
      url: '/api/doc/export_upload_info',
      method: 'POST',
      data: {
        dentry_uuid: dentryUuid,
        export_format: 'pdf',
      },
    });

    // 步骤2: createExportJob (PDF使用不同的API)
    const job = await lwpClient.request({
      url: '/api/doc/create_export_job',
      method: 'POST',
      data: {
        export_info: uploadInfo,
      },
    });

    // 步骤3: queryExportStatus - 轮询状态
    let status = 'pending';
    let downloadUrl = '';
    const pollInterval = 2000;

    while (status === 'pending' || status === 'processing') {
      const result = await lwpClient.request({
        url: '/api/doc/query_export_status',
        method: 'POST',
        data: {
          job_id: job.job_id,
        },
      });

      status = result.status;
      
      if (status === 'success') {
        downloadUrl = result.download_url;
        break;
      } else if (status === 'failed') {
        throw new Error(result.error_message || '导出失败');
      }

      await new Promise(resolve => setTimeout(resolve, pollInterval));
    }

    return downloadUrl;
  }

  /**
   * 导出白板为图片
   * 参考老项目 api.js 中的 downloadBoard 方法
   */
  async exportBoard(dentryUuid: string): Promise<string> {
    const lwpClient = (window as any).lwpClient;

    const response = await lwpClient.request({
      url: '/api/board/export',
      method: 'POST',
      data: {
        dentry_uuid: dentryUuid,
        format: 'jpg',
      },
    });

    return response.download_url;
  }

  /**
```

/**
 * 钉钉文档内容块类型

interface ContentBlock {
  type: 'text' | 'heading' | 'code' | 'image' | 'list' | 'table' | 'quote' | 'divider';
  level?: number;      // heading的级别
  text?: string;       // 文本内容
  language?: string;  // 代码块语言
  code?: string;      // 代码内容
  alt?: string;       // 图片描述
  attachment_id?: string; // 附件ID
  items?: ContentBlock[]; // 列表项
  rows?: string[][];   // 表格行
}

/**
 * 钉钉文档结构
 */
interface DingDocument {
  dentry_uuid: string;
  dentry_name: string;
  dentry_type: string;
  content?: {
    blocks: ContentBlock[];
  };
}

/**
 * 将钉钉文档转换为Markdown
 * 参考老项目 adoc2md.js 的转换逻辑
 */
export function convertAdocToMarkdown(doc: DingDocument): string {
  let markdown = '';
  
  // 添加文档标题
  markdown += `# ${doc.dentry_name}\n\n`;
  
  // 处理文档内容
  if (doc.content && doc.content.blocks) {
    markdown += processBlocks(doc.content.blocks);
  }
  
  return markdown;
}

/**
 * 处理文档内容块
 */
function processBlocks(blocks: ContentBlock[]): string {
  let result = '';
  
  for (const block of blocks) {
    switch (block.type) {
      case 'text':
        // 普通文本
        result += block.text + '\n\n';
        break;
        
      case 'heading':
        // 标题
        const level = '#'.repeat(block.level || 1);
        result += `${level} ${block.text}\n\n`;
        break;
        
      case 'code':
        // 代码块
        const language = block.language || '';
        result += '```' + language + '\n';
        result += (block.code || '') + '\n';
        result += '```\n\n';
        break;
        
      case 'image':
        // 图片
        const alt = block.alt || '图片';
        const attachmentRef = `attachment_id?=${block.attachment_id}`;
        result += `![${alt}](${attachmentRef})\n\n`;
        break;
        
      case 'list':
        // 列表
        if (block.items) {
          for (const item of block.items) {
            result += processBlocks([item]);
          }
        }
        result += '\n';
        break;
        
      case 'table':
        // 表格
        if (block.rows && block.rows.length > 0) {
          // 表头
          result += '| ' + block.rows[0].join(' | ') + ' |\n';
          result += '| ' + block.rows[0].map(() => '---').join(' | ') + ' |\n';
          // 表格内容
          for (let i = 1; i < block.rows.length; i++) {
            result += '| ' + block.rows[i].join(' | ') + ' |\n';
          }
        }
        result += '\n';
        break;
        
      case 'quote':
        // 引用
        if (block.text) {
          result += `> ${block.text}\n\n`;
        }
        break;
        
      case 'divider':
        // 分割线
        result += '---\n\n';
        break;
        
      default:
        // 未知类型，保留原始数据
        console.warn('未知的块类型:', block.type);
    }
  }
  
  return result;
}

/**
 * 清理Markdown文本
 */
export function cleanMarkdown(markdown: string): string {
  // 移除多余的空行
  markdown = markdown.replace(/\n{3,}/g, '\n\n');
  
  // 移除行尾空格
  markdown = markdown.replace(/ +\n/g, '\n');
  
  return markdown.trim();
}

```typescript
export function getFileExtension(filename: string): string {
  const parts = filename.split('.');
  return parts.length > 1 ? parts[parts.length - 1] : '';
}

/**
 * 更改文件扩展名
 */
export function changeExtension(filename: string, newExt: string): string {
  const parts = filename.split('.');
  if (parts.length > 1) {
    parts[parts.length - 1] = newExt;
    return parts.join('.');
  }
  return filename + '.' + newExt;
}

/**
 * 生成安全的文件名
 */
export function sanitizeFilename(filename: string): string {
  // 移除或替换不安全的文件名字符
  return filename
    .replace(/[<>:"/\\|?*]/g, '') // 移除不安全字符
    .replace(/\s+/g, '_')            // 空格替换为下划线
    .substring(0, 200);              // 限制长度
}

/**
 * 处理文件名冲突
 */
export function resolveDuplicateFilename(
  existingFiles: string[],
  filename: string,
  maxAttempts = 100
): string {
  if (!existingFiles.includes(filename)) {
    return filename;
  }
  
  const nameWithoutExt = filename.substring(0, filename.lastIndexOf('.'));
  const ext = filename.substring(filename.lastIndexOf('.'));
  
  for (let i = 1; i < maxAttempts; i++) {
    const newFilename = `${nameWithoutExt} (${i})${ext}`;
    if (!existingFiles.includes(newFilename)) {
      return newFilename;
    }
  }
  
  // 如果还是冲突，返回带时间戳的文件名
  return `${nameWithoutExt}_${Date.now()}${ext}`;
}
```

```typescript
import { ref, computed } from 'vue';
import type { DentryItem } from '@/sidepanel/api/types';

interface ExportTask {
  id: string;
  item: DentryItem;
  status: 'pending' | 'exporting' | 'success' | 'failed';
  progress: number;
  error?: string;
  downloadUrl?: string;
}

export function useExportManager(concurrentLimit = 3) {
  const queue = ref<ExportTask[]>([]);
  const isExporting = computed(() => 
    queue.value.some(t => t.status === 'exporting')
  );

  function addTask(item: DentryItem): void {
    const task: ExportTask = {
      id: `${item.dentry_uuid}_${Date.now()}`,
      item,
      status: 'pending',
      progress: 0,
    };
    queue.value.push(task);
    processQueue();
  }

  function addTasks(items: DentryItem[]): void {
    for (const item of items) {
      addTask(item);
    }
  }

  async function processQueue(): Promise<void> {
    const pendingTasks = queue.value.filter(t => t.status === 'pending');
    const runningTasks = queue.value.filter(t => t.status === 'exporting');

    if (runningTasks.length >= concurrentLimit) {
      return; // 已达并发上限
    }

    const tasksToStart = pendingTasks.slice(0, concurrentLimit - runningTasks.length);
    
    for (const task of tasksToStart) {
      exportTask(task);
    }
  }

  async function exportTask(task: ExportTask): Promise<void> {
    task.status = 'exporting';
    
    try {
      // 调用导出API
      // const api = new DingTalkAPI();
      // const url = await api.exportDocument(task.item.dentry_uuid, 'md');
      // task.downloadUrl = url;
      
      task.status = 'success';
      task.progress = 100;
    } catch (e) {
      task.status = 'failed';
      task.error = (e as Error).message;
    }
  }

  function retryTask(taskId: string): void {
    const task = queue.value.find(t => t.id === taskId);
    if (task && task.status === 'failed') {
      task.status = 'pending';
      task.progress = 0;
      task.error = undefined;
      processQueue();
    }
  }

  function removeTask(taskId: string): void {
    const index = queue.value.findIndex(t => t.id === taskId);
    if (index !== -1) {
      queue.value.splice(index, 1);
    }
  }

  function clearQueue(): void {
    queue.value = queue.value.filter(t => t.status !== 'success');
  }

  return {
    queue,
    isExporting,
    addTask,
    addTasks,
    retryTask,
    removeTask,
    clearQueue,
  };
}
```

## 6. 错误处理步骤

### 6.1 错误提示组件

**步骤1：创建Toast组件**

在 `src/sidepanel/components/common/Toast/` 目录下创建 `index.vue`：

```vue
<template>
  <Transition name="toast">
    <div 
      v-if="visible"
      :class="toastClasses"
    >
      <span class="mr-2">{{ icon }}</span>
      <span>{{ message }}</span>
      <button @click="visible = false" class="ml-4 text-white hover:text-gray-200">
        ✕
      </button>
    </div>
  </Transition>
</template>

<script setup lang="ts">
import { ref, computed, watch } from 'vue';

interface Props {
  modelValue: boolean;
  message: string;
  type?: 'success' | 'error' | 'warning' | 'info';
  duration?: number;
}

const props = withDefaults(defineProps<Props>(), {
  type: 'info',
  duration: 3000,
});

const emit = defineEmits<{
  (e: 'update:modelValue', value: boolean): void;
}>();

const visible = ref(props.modelValue);

watch(() => props.modelValue, (val) => {
  visible.value = val;
});

watch(visible, (val) => {
  emit('update:modelValue', val);
  
  if (val && props.duration > 0) {
    setTimeout(() => {
      visible.value = false;
    }, props.duration);
  }
});

const toastClasses = computed(() => {
  const base = 'fixed top-4 right-4 px-4 py-3 rounded-lg shadow-lg text-white z-50';
  
  const types = {
    success: 'bg-success',
    error: 'bg-error',
    warning: 'bg-warning',
    info: 'bg-primary',
  };
  
  return [
    base,
    types[props.type],
  ];
});

const icon = computed(() => {
  const icons = {
    success: '✅',
    error: '❌',
    warning: '⚠️',
    info: 'ℹ️',
  };
  return icons[props.type];
});
</script>
```

**步骤2：创建useToast Composable**

在 `src/sidepanel/composables/` 目录下创建 `useToast.ts`：

```typescript
import { ref } from 'vue';

export function useToast() {
  const toasts = ref<Array<{
    id: string;
    message: string;
    type: 'success' | 'error' | 'warning' | 'info';
  }>>([]);

  function show(message: string, type: 'success' | 'error' | 'warning' | 'info' = 'info') {
    const id = Date.now().toString();
    toasts.value.push({ id, message, type });
    
    setTimeout(() => {
      remove(id);
    }, 3000);
  }

  function remove(id: string) {
    const index = toasts.value.findIndex(t => t.id === id);
    if (index !== -1) {
      toasts.value.splice(index, 1);
    }
  }

  return {
    toasts,
    show,
    success: (message: string) => show(message, 'success'),
    error: (message: string) => show(message, 'error'),
    warning: (message: string) => show(message, 'warning'),
    info: (message: string) => show(message, 'info'),
  };
}
```

## 8. 常见问题与解决方案

### 8.1 开发环境问题

**问题1：Vite构建报错**

```
解决方案：
1. 清除node_modules和lock文件：rm -rf node_modules package-lock.json
2. 重新安装：npm install
3. 清除Vite缓存：rm -rf .vite
```

**问题2：TypeScript类型错误**

```
解决方案：
1. 检查tsconfig.json配置
2. 确保已安装@types/chrome
3. 检查路径别名配置
```

### 8.2 Chrome Extension问题

**问题1：扩展无法加载**

```
解决方案：
1. 检查manifest.json语法
2. 确保manifest_version为3
3. 检查权限配置是否正确
```

**问题2：Content Script注入失败**

```
解决方案：
1. 检查matches规则是否正确
2. 确保run_at时机正确
3. 检查控制台是否有错误信息
```

### 8.3 钉钉API问题

**问题1：API调用失败**

```
解决方案：
1. 检查网络请求是否被CORS阻止
2. 确认是否在钉钉页面环境下
3. 检查Cookie是否正确传递
```

**问题2：导出任务卡住**

```
解决方案：
1. 检查轮询逻辑是否正确
2. 添加超时机制
3. 实现重试机制
```
