```typescript
import { cosmiconfig } from 'cosmiconfig';
import path from 'path';
import os from 'os';

export interface ConfigProfile {
  outputDir: string;
  concurrency: number;
  downloadImages: boolean;
  downloadAssets: boolean;
  incrementalMode: boolean;
}

export interface Config {
  profiles: Record<string, ConfigProfile>;
  currentProfile: string;
}

export class ConfigManager {
  private configDir: string;
  private explorer: any;

  constructor() {
    this.configDir = path.join(os.homedir(), '.dingding-export');
    this.explorer = cosmiconfig('dingding-export', {
      searchPlaces: [
        '.dingding-exportrc',
        '.dingding-exportrc.json',
        'package.json',
      ],
    });
  }

  async load(): Promise<Config> {
    const result = await this.explorer.search();
    return result?.config || this.getDefaultConfig();
  }

  async getProfile(name?: string): Promise<ConfigProfile> {
    const config = await this.load();
    const profileName = name || config.currentProfile;
    return config.profiles[profileName] || config.profiles.default;
  }

  private getDefaultConfig(): Config {
    return {
      profiles: {
        default: {
          outputDir: './output',
          concurrency: 5,
          downloadImages: true,
          downloadAssets: true,
          incrementalMode: false,
        },
      },
      currentProfile: 'default',
    };
  }
}
```

### 步骤 2.3：实现 Cookie 管理 ⭐

#### 2.3.1 创建 Cookie 基础结构

创建 `src/core/CookieManager.ts`：

```typescript
import { CookieJar } from 'tough-cookie';
import { chromium } from 'playwright';
import path from 'path';
import os from 'os';
import fs from 'fs/promises';

export interface Cookie {
  name: string;
  value: string;
  domain: string;
  path: string;
  expires?: string;
}

export class CookieManager {
  private cookiePath: string;
  private cookieJar: CookieJar;

  constructor() {
    this.cookiePath = path.join(os.homedir(), '.dingding-export', 'cookies.json');
    this.cookieJar = new CookieJar();
  }

  /**
   * 验证 Cookie 是否有效
   */
  async validate(): Promise<boolean> {
    try {
      await this.loadCookie();
      return true;
    } catch {
      return false;
    }
  }

  /**
   * 获取 Cookie 字符串
   */
  async getCookieString(): Promise<string> {
    const cookies = await this.loadCookie();
    return cookies
      .map((c: Cookie) => `${c.name}=${c.value}`)
      .join('; ');
  }

  /**
   * 使用 Playwright 自动登录
   */
  async refreshWithPlaywright(): Promise<void> {
    const browser = await chromium.launch({ headless: false });
    const context = await browser.newContext();
    const page = await context.newPage();

    try {
      // 导航到钉钉登录页
      await page.goto('https://login.dingtalk.com');

      console.log('请在浏览器中完成登录...');
      
      // 等待用户登录（等待跳转到首页）
      await page.waitForURL('https://alidocs.dingtalk.com/**', { timeout: 0 });

      // 获取 Cookie
      const cookies = await context.cookies();
      
      // 保存 Cookie
      await this.saveCookie(cookies);
      
      console.log('Cookie 已保存');
    } finally {
      await browser.close();
    }
  }

  private async loadCookie(): Promise<Cookie[]> {
    const content = await fs.readFile(this.cookiePath, 'utf-8');
    return JSON.parse(content);
  }

  private async saveCookie(cookies: any[]): Promise<void> {
    const cookieDir = path.dirname(this.cookiePath);
    await fs.mkdir(cookieDir, { recursive: true });
    await fs.writeFile(this.cookiePath, JSON.stringify(cookies, null, 2));
  }
}
```

```typescript
import * as cheerio from 'cheerio';
import { HttpClient } from './HttpClient';
import { CookieManager } from '../core/CookieManager';

export class DingTalkAPI {
  private http: HttpClient;
  private cookieManager: CookieManager;

  constructor() {
    this.http = new HttpClient();
    this.cookieManager = new CookieManager();
  }

  /**
   * 获取文档页面（提取 dentryKey）
   */
  async fetchDocumentPage(nodeId: string): Promise<string> {
    await this.injectCookie();
    const response = await this.http.get(`/i/nodes/${nodeId}`);
    return response.data;
  }

  /**
   * 提取 dentryKey
   */
  extractDentryKey(html: string): string {
    const $ = cheerio.load(html);
    const scriptContent = $('#mainsite_server_content').html();
    
    if (!scriptContent) {
      throw new Error('无法找到 mainsite_server_content');
    }

    const data = JSON.parse(scriptContent);
    return data.dentryKey;
  }

  /**
   * 获取文档内容
   */
  async fetchDocumentData(dentryKey: string): Promise<any> {
    await this.injectCookie();
    
    const response = await this.http.post(
      '/api/document/data',
      { dentryKey },
      {
        'a-dentry-key': dentryKey,
        'Content-Type': 'application/json',
      }
    );

    return response.data;
  }

  /**
   * 下载图片
   */
  async downloadImage(url: string): Promise<Buffer> {
    const response = await this.http.get(url, { responseType: 'arraybuffer' });
    return response.data;
  }

  private async injectCookie() {
    const cookieString = await this.cookieManager.getCookieString();
    this.http.setCookie(cookieString);
  }
}
```

### 步骤 2.6：实现 Markdown 生成

创建 `src/generators/MarkdownGenerator.ts`：

```typescript
import { ParsedDocument } from '../parsers/DocumentParser';

export class MarkdownGenerator {
  generate(document: ParsedDocument): string {
    let markdown = `# ${document.title}\n\n`;

    for (const content of document.content) {
      markdown += this.generateContent(content);
    }

    return markdown;
  }

  private generateContent(content: any): string {
    switch (content.type) {
      case 'heading':
        return `${'#'.repeat(content.level || 1)} ${content.text}\n\n`;
      case 'paragraph':
        return `${content.text}\n\n`;
      case 'code':
        return `\`\`\`${content.language || ''}\n${content.text}\n\`\`\`\n\n`;
      case 'table':
        return this.generateTable(content.rows);
      default:
        return '';
    }
  }

  private generateTable(rows?: string[][]): string {
    if (!rows || rows.length === 0) return '';

    let table = '';
    for (const row of rows) {
      table += '| ' + row.join(' | ') + ' |\n';
    }
    return table + '\n';
  }
}
```

```typescript
import { DingTalkAPI } from '../api/DingTalkAPI';
import { DocumentParser } from '../parsers/DocumentParser';
import { MarkdownGenerator } from '../generators/MarkdownGenerator';
import fs from 'fs/promises';
import path from 'path';

export class ExportService {
  private api: DingTalkAPI;
  private parser: DocumentParser;
  private generator: MarkdownGenerator;

  constructor() {
    this.api = new DingTalkAPI();
    this.parser = new DocumentParser();
    this.generator = new MarkdownGenerator();
  }

  async exportDocument(docId: string, outputDir: string): Promise<void> {
    console.log(`正在导出文档: ${docId}`);

    // 1. 获取文档页面
    const html = await this.api.fetchDocumentPage(docId);

    // 2. 提取 dentryKey
    const dentryKey = this.api.extractDentryKey(html);

    // 3. 获取文档内容
    const documentData = await this.api.fetchDocumentData(dentryKey);

    // 4. 解析文档
    const parsedDoc = this.parser.parse(documentData);

    // 5. 生成 Markdown
    const markdown = this.generator.generate(parsedDoc);

    // 6. 保存文件
    await fs.mkdir(outputDir, { recursive: true });
    const outputPath = path.join(outputDir, `${docId}.md`);
    await fs.writeFile(outputPath, markdown, 'utf-8');

    console.log(`导出成功: ${outputPath}`);
  }
}
```

```typescript
import fs from 'fs/promises';
import path from 'path';

export class StateManager {
  private statePath: string;

  constructor(outputDir: string) {
    this.statePath = path.join(outputDir, '.state.json');
  }

  async loadState(): Promise<Record<string, string>> {
    try {
      const content = await fs.readFile(this.statePath, 'utf-8');
      return JSON.parse(content);
    } catch {
      return {};
    }
  }

  async saveState(state: Record<string, string>): Promise<void> {
    await fs.writeFile(this.statePath, JSON.stringify(state, null, 2));
  }

  async shouldExport(docId: string, lastModified: string): Promise<boolean> {
    const state = await this.loadState();
    const lastExport = state[docId];
    return !lastExport || lastModified > lastExport;
  }

  async markExported(docId: string, timestamp: string): Promise<void> {
    const state = await this.loadState();
    state[docId] = timestamp;
    await this.saveState(state);
  }
}
```

### 步骤 3.2：实现批量导出

创建 `src/cli/commands/docs.ts`：

```typescript
import { Command } from 'commander';
import { ExportService } from '../../core/ExportService';
import fs from 'fs/promises';
import pLimit from 'p-limit';
import cliProgress from 'cli-progress';

export const docsCommand = new Command('docs')
  .description('批量导出文档')
  .option('--file <path>', '文档 ID 列表文件')
  .option('--ids <list>', '逗号分隔的文档 ID')
  .option('-o, --output <dir>', '输出目录', './output')
  .option('-c, --concurrency <n>', '并发数量', '5')
  .action(async (options) => {
    const service = new ExportService();
    
    // 获取文档 ID 列表
    let docIds: string[] = [];
    if (options.file) {
      const content = await fs.readFile(options.file, 'utf-8');
      docIds = content.split('\n').filter(Boolean);
    } else if (options.ids) {
      docIds = options.ids.split(',');
    } else {
      console.error('请指定 --file 或 --ids');
      process.exit(1);
    }

    // 并发控制
    const limit = pLimit(parseInt(options.concurrency));
    const progressBar = new cliProgress.SingleBar(
      {},
      cliProgress.Presets.shades_classic
    );

    progressBar.start(docIds.length, 0);

    const tasks = docIds.map((docId) =>
      limit(async () => {
        await service.exportDocument(docId, options.output);
        progressBar.increment();
      })
    );

    await Promise.all(tasks);
    progressBar.stop();
  });
```
